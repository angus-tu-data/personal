    input_label_cols=['price']

sql = """
select
  date_add(date '2020-01-01', interval days day) AS date,
  predicted_price as price
from ML.PREDICT(
  model `financial_reports_yahoo.stock_model`,
  (select date_diff(d, date '2020-01-01', day) AS days
   from unnest(generate_date_array('2020-01-01', '2025-12-25')) as d)
)
"""
dfp = BQ.query(sql).to_dataframe()

sql = """
    create or replace model `financial_reports_yahoo.stock_ts`
    options(
    model_type='ARIMA_PLUS',
    time_series_timestamp_col='date',
    time_series_data_col='price'
    ) 
    as
    select *
    from `financial_reports_yahoo.stock_price_daily_bqml`
"""

job = BQ.query(sql)
job.result()

sql = """
select date(forecast_timestamp) as date, forecast_value as price
from ML.FORECAST(
  model `financial_reports_yahoo.stock_ts`,
  struct(30 as horizon)
)
"""
dfts = BQ.query(sql).to_dataframe()
dfts = dfts.sort_values('date')
